11class CollBase(object): 

12 """ 

13 Generic collocation class, that contains everything to do integration over 

14 intervals and between nodes. 

15 It can be used to produce many kind of quadrature nodes from various 

16 distribution (awesome!). 

17 

18 It is based on the two main parameters that define the nodes : 

19 

20 - node_type : the node distribution used for the collocation method 

21 - quad_type : the type of quadrature used (inclusion of not of boundary) 

22 

23 Current implementation provides the following available parameter values 

24 for node_type : 

25 

26 - EQUID : equidistant node distribution 

27 - LEGENDRE : distribution from Legendre polynomials 

28 - CHEBY-{1,2,3,4} : distribution from Chebyshev polynomials of a given kind 

29 

30 The type of quadrature cann be GAUSS (only inner nodes), RADAU-LEFT 

31 (inclusion of the left boundary), RADAU-RIGHT (inclusion of the right 

32 boundary) and LOBATTO (inclusion of left and right boundary). 

33 

34 Furthermore, the ``useSpline`` option can be activated to eventually use 

35 spline interpolation when computing the weights. 

36 

37 Here is the equivalency table with the original classes implemented in 

38 pySDC : 

39 

40 +-------------------------+-----------+-------------+-----------+ 

41 | Original Class | node_type | quad_type | useSpline | 

42 +=========================+===========+=============+===========+ 

43 | Equidistant | EQUID | LOBATTO | False | 

44 +-------------------------+-----------+-------------+-----------+ 

45 | EquidistantInner | EQUID | GAUSS | False | 

46 +-------------------------+-----------+-------------+-----------+ 

47 | EquidistantNoLeft | EQUID | RADAU-RIGHT | False | 

48 +-------------------------+-----------+-------------+-----------+ 

49 | EquidistantSpline_Right | EQUID | RADAU-RIGHT | True | 

50 +-------------------------+-----------+-------------+-----------+ 

51 | CollGaussLegendre | LEGENDRE | GAUSS | False | 

52 +-------------------------+-----------+-------------+-----------+ 

53 | CollGaussLobatto | LEGENDRE | LOBATTO | False | 

54 +-------------------------+-----------+-------------+-----------+ 

55 | CollGaussRadau_Left | LEGENDRE | RADAU-LEFT | False | 

56 +-------------------------+-----------+-------------+-----------+ 

57 | CollGaussRadau_Right | LEGENDRE | RADAU-RIGHT | False | 

58 +-------------------------+-----------+-------------+-----------+ 

59 

60 Attributes: 

61 num_nodes (int): number of collocation nodes 

62 tleft (float): left interval point 

63 tright (float): right interval point 

64 nodes (numpy.ndarray): array of quadrature nodes 

65 weights (numpy.ndarray): array of quadrature weights for the full interval 

66 Qmat (numpy.ndarray): matrix containing the weights for tleft to node 

67 Smat (numpy.ndarray): matrix containing the weights for node to node 

68 delta_m (numpy.ndarray): array of distances between nodes 

69 right_is_node (bool): flag to indicate whether right point is collocation node 

70 left_is_node (bool): flag to indicate whether left point is collocation node 

71 """ 

72 

73 def __init__( 

74 self, num_nodes=None, tleft=0, tright=1, node_type='LEGENDRE', quad_type=None, useSpline=False, **kwargs 

75 ): 

76 """ 

77 Initialization routine for a collocation object 

78 

79 Args: 

80 num_nodes (int): number of collocation nodes 

81 tleft (float): left interval point 

82 tright (float): right interval point 

83 """ 

84 

85 if not num_nodes > 0: 

86 raise CollocationError('At least one quadrature node required, got %s' % num_nodes) 

87 if not tleft < tright: 

88 raise CollocationError('Interval boundaries are corrupt, got %s and %s' % (tleft, tright)) 

89 

90 self.logger = logging.getLogger('collocation') 

91 

92 # Set number of nodes, left and right interval boundaries 

93 self.num_nodes = num_nodes 

94 self.tleft = tleft 

95 self.tright = tright 

96 

97 self.node_type = node_type 

98 self.quad_type = quad_type 

99 

100 # Instanciate attributes 

101 self.nodeGenerator = NodesGenerator(self.node_type, self.quad_type) 

102 if useSpline: 

103 self._getWeights = self._getWeights_spline 

104 # We need: 1<=order<=5 and order < num_nodes 

105 self.order = min(num_nodes - 1, 3) 

106 elif self.node_type == 'EQUID': 

107 self.order = num_nodes 

108 else: 

109 if self.quad_type == 'GAUSS': 

110 self.order = 2 * num_nodes 

111 elif self.quad_type.startswith('RADAU'): 

112 self.order = 2 * num_nodes - 1 

113 elif self.quad_type == 'LOBATTO': 

114 self.order = 2 * num_nodes - 2 

115 

116 self.left_is_node = self.quad_type in ['LOBATTO', 'RADAU-LEFT'] 

117 self.right_is_node = self.quad_type in ['LOBATTO', 'RADAU-RIGHT'] 

118 

119 self.nodes = self._getNodes 

120 self.weights = self._getWeights(tleft, tright) 

121 self.Qmat = self._gen_Qmatrix_spline if useSpline else self._gen_Qmatrix 

122 self.Smat = self._gen_Smatrix 

123 self.delta_m = self._gen_deltas

125 @staticmethod 

126 def evaluate(weights, data): 

127 """ 

128 Evaluates the quadrature over the full interval 

129 

130 Args: 

131 weights (numpy.ndarray): array of quadrature weights for the full interval 

132 data (numpy.ndarray): f(x) to be integrated 

133 

134 Returns: 

135 numpy.ndarray: integral over f(x) between tleft and tright 

136 """ 

137 if not np.size(weights) == np.size(data): 

138 raise CollocationError("Input size does not match number of weights, but is %s" % np.size(data)) 

139 

140 return np.dot(weights, data)

142 def _getWeights(self, a, b): 

143 """ 

144 Computes weights using barycentric interpolation 

145 

146 Args: 

147 a (float): left interval boundary 

148 b (float): right interval boundary 

149 

150 Returns: 

151 numpy.ndarray: weights of the collocation formula given by the nodes 

152 """ 

153 if self.nodes is None: 

154 raise CollocationError(f"Need nodes before computing weights, got {self.nodes}") 

155 

156 # Instantiate the Lagrange interpolator object 

157 approx = LagrangeApproximation(self.nodes, weightComputation='AUTO') 

158 

159 # Compute weights 

160 tLeft = np.ravel(self.tleft)[0] 

161 tRight = np.ravel(self.tright)[0] 

162 weights = approx.getIntegrationMatrix([(tLeft, tRight)], numQuad='FEJER') 

163 

164 return np.ravel(weights)

166 @property 

167 def _getNodes(self): 

168 """ 

169 Computes nodes using an internal NodesGenerator object 

170 

171 Returns: 

172 np.ndarray: array of Gauss-Legendre nodes 

173 """ 

174 # Generate normalized nodes in [-1, 1] 

175 nodes = self.nodeGenerator.getNodes(self.num_nodes) 

176 

177 # Scale nodes to [tleft, tright] 

178 a = self.tleft 

179 b = self.tright 

180 nodes += 1.0 

181 nodes /= 2.0 

182 nodes *= b - a 

183 nodes += a 

184 

185 if self.left_is_node: 

186 nodes[0] = self.tleft 

187 if self.right_is_node: 

188 nodes[-1] = self.tright 

189 

190 return nodes

192 @property 

193 def _gen_Qmatrix(self): 

194 """ 

195 Compute tleft-to-node integration matrix for later use in collocation formulation 

196 

197 Returns: 

198 numpy.ndarray: matrix containing the weights for tleft to node 

199 """ 

200 if self.nodes is None: 

201 raise CollocationError(f"Need nodes before computing weights, got {self.nodes}") 

202 M = self.num_nodes 

203 Q = np.zeros([M + 1, M + 1]) 

204 

205 # Instantiate the Lagrange interpolator object 

206 approx = LagrangeApproximation(self.nodes, weightComputation='AUTO') 

207 

208 # Compute tleft-to-node integration matrix 

209 tLeft = np.ravel(self.tleft)[0] 

210 intervals = [(tLeft, tau) for tau in self.nodes] 

211 intQ = approx.getIntegrationMatrix(intervals, numQuad='FEJER') 

212 

213 # Store into Q matrix 

214 Q[1:, 1:] = intQ 

215 

216 return Q

218 @property 

219 def _gen_Smatrix(self): 

220 """ 

221 Compute node-to-node integration matrix for later use in collocation formulation 

222 

223 Returns: 

224 numpy.ndarray: matrix containing the weights for node to node 

225 """ 

226 M = self.num_nodes 

227 Q = self.Qmat 

228 S = np.zeros([M + 1, M + 1]) 

229 

230 S[1, :] = Q[1, :] 

231 for m in np.arange(2, M + 1): 

232 S[m, :] = Q[m, :] - Q[m - 1, :] 

233 

234 return S

236 @property 

237 def _gen_deltas(self): 

238 """ 

239 Compute distances between the nodes 

240 

241 Returns: 

242 numpy.ndarray: distances between the nodes 

243 """ 

244 M = self.num_nodes 

245 delta = np.zeros(M) 

246 delta[0] = self.nodes[0] - self.tleft 

247 for m in np.arange(1, M): 

248 delta[m] = self.nodes[m] - self.nodes[m - 1] 

249 

250 return delta

252 def _getWeights_spline(self, a, b): 

253 """ 

254 Computes weights using spline interpolation instead of Gaussian 

255 quadrature 

256 

257 Args: 

258 a (float): left interval boundary 

259 b (float): right interval boundary 

260 

261 Returns: 

262 np.ndarray: weights of the collocation formula given by the nodes 

263 """ 

264 

265 # get the defining tck's for each spline basis function 

266 circ_one = np.zeros(self.num_nodes) 

267 circ_one[0] = 1.0 

268 tcks = [] 

269 for i in range(self.num_nodes): 

270 tcks.append( 

271 intpl.splrep(self.nodes, np.roll(circ_one, i), xb=self.tleft, xe=self.tright, k=self.order, s=0.0) 

272 ) 

273 

274 weights = np.zeros(self.num_nodes) 

275 for i in range(self.num_nodes): 

276 weights[i] = intpl.splint(a, b, tcks[i]) 

277 

278 return weights

280 @property 

281 def _gen_Qmatrix_spline(self): 

282 """ 

283 Compute tleft-to-node integration matrix for later use in collocation formulation 

284 Returns: 

285 numpy.ndarray: matrix containing the weights for tleft to node 

286 """ 

287 M = self.num_nodes 

288 Q = np.zeros([M + 1, M + 1]) 

289 

290 # for all nodes, get weights for the interval [tleft,node] 

291 for m in np.arange(M): 

292 Q[m + 1, 1:] = self._getWeights(self.tleft, self.nodes[m]) 

293 

294 return Q
